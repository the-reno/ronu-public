Option Explicit

'====================================================================
' DAILY TIME-DEPOSIT ACCRUAL ENGINE - ALL TENORS
'
' Excel 365 formula:
'   =r_DailyAccrual(H2,H3,H4,A:F)
'
' Inputs:
'   AnalysisStart      Requested analysis start date
'   AnalysisEnd        Analysis end date, inclusive
'   InitialNotional    Initial cash invested in EACH tenor simulation
'   CurveRange         Curve table including headers
'
' Required curve headers:
'   Date | ON | 1M | 2M | 3M | 6M
'
' Conventions:
'   - Rates are always entered and returned as percentages.
'     Example: 1.54 means 1.54%.
'   - ACT/360 simple interest.
'   - Interest is always reinvested at each roll.
'   - Missing dates or rates use the latest valid prior curve date.
'   - Monthly roll targets remain anchored to the original requested
'     start date. This avoids date drift.
'   - The output is a long-form table containing all five tenors.
'====================================================================

Public Function r_DailyAccrual( _
    ByVal AnalysisStart As Variant, _
    ByVal AnalysisEnd As Variant, _
    ByVal InitialNotional As Variant, _
    ByVal CurveRange As Range) As Variant

    On Error GoTo CleanFail

    Dim sourceRange As Range
    Dim curveData As Variant
    Dim tenors As Variant
    Dim tenorColumns(0 To 4) As Long
    Dim initialDates(0 To 4) As Date
    Dim initialRates(0 To 4) As Double

    Dim dateCol As Long
    Dim validationMessage As String
    Dim requestedStart As Date
    Dim analysisEndDate As Date
    Dim startingPrincipal As Double

    Dim totalRows As Long
    Dim outputRow As Long
    Dim tenorIndex As Long
    Dim result() As Variant
    Dim buildError As String

    '-----------------------------
    ' Validate formula inputs
    '-----------------------------
    If Not TryGetExcelDate(AnalysisStart, requestedStart) Then
        r_DailyAccrual = ErrorResult("Analysis start is not a valid date.")
        Exit Function
    End If

    If Not TryGetExcelDate(AnalysisEnd, analysisEndDate) Then
        r_DailyAccrual = ErrorResult("Analysis end is not a valid date.")
        Exit Function
    End If

    If requestedStart > analysisEndDate Then
        r_DailyAccrual = ErrorResult("Analysis start must be on or before analysis end.")
        Exit Function
    End If

    If Not IsNumeric(InitialNotional) Then
        r_DailyAccrual = ErrorResult("Initial notional must be numeric.")
        Exit Function
    End If

    startingPrincipal = CDbl(InitialNotional)
    If startingPrincipal <= 0 Then
        r_DailyAccrual = ErrorResult("Initial notional must be greater than zero.")
        Exit Function
    End If

    If CurveRange Is Nothing Then
        r_DailyAccrual = ErrorResult("Curve range was not provided.")
        Exit Function
    End If

    'Trim a whole-column reference such as A:F to the worksheet used range.
    Set sourceRange = Application.Intersect(CurveRange, CurveRange.Worksheet.UsedRange)
    If sourceRange Is Nothing Then
        r_DailyAccrual = ErrorResult("Curve range does not contain data.")
        Exit Function
    End If

    If sourceRange.Rows.Count < 2 Or sourceRange.Columns.Count < 6 Then
        r_DailyAccrual = ErrorResult("Curve range must include headers and at least one data row.")
        Exit Function
    End If

    curveData = sourceRange.Value2
    tenors = Array("ON", "1M", "2M", "3M", "6M")

    dateCol = FindHeaderColumn(curveData, "DATE")
    If dateCol = 0 Then
        r_DailyAccrual = ErrorResult("The curve table does not contain a Date header.")
        Exit Function
    End If

    For tenorIndex = 0 To 4
        tenorColumns(tenorIndex) = FindHeaderColumn(curveData, CStr(tenors(tenorIndex)))
        If tenorColumns(tenorIndex) = 0 Then
            r_DailyAccrual = ErrorResult("Missing curve header: " & CStr(tenors(tenorIndex)) & ".")
            Exit Function
        End If
    Next tenorIndex

    validationMessage = ValidateCurveData(curveData, dateCol, tenorColumns)
    If Len(validationMessage) > 0 Then
        r_DailyAccrual = ErrorResult(validationMessage)
        Exit Function
    End If

    '-----------------------------
    ' Find the initial date and rate for every tenor.
    ' The prior-date convention is applied independently by tenor.
    '-----------------------------
    totalRows = 1        'Header row

    For tenorIndex = 0 To 4
        If Not GetPointOnOrBefore( _
                curveData, _
                dateCol, _
                tenorColumns(tenorIndex), _
                requestedStart, _
                initialDates(tenorIndex), _
                initialRates(tenorIndex)) Then

            r_DailyAccrual = ErrorResult( _
                "No valid " & CStr(tenors(tenorIndex)) & _
                " rate exists on or before the analysis start date.")
            Exit Function
        End If

        If initialDates(tenorIndex) > analysisEndDate Then
            r_DailyAccrual = ErrorResult( _
                "The adjusted start date for " & CStr(tenors(tenorIndex)) & _
                " is after the analysis end date.")
            Exit Function
        End If

        totalRows = totalRows + DateDiff("d", initialDates(tenorIndex), analysisEndDate) + 1
    Next tenorIndex

    If totalRows > 1048576 Then
        r_DailyAccrual = ErrorResult("The result exceeds Excel's worksheet row limit.")
        Exit Function
    End If

    ReDim result(1 To totalRows, 1 To 18)
    WriteHeaders result
    outputRow = 2

    '-----------------------------
    ' Simulate each tenor independently.
    ' Output order: ON, 1M, 2M, 3M, 6M.
    '-----------------------------
    For tenorIndex = 0 To 4
        If Not BuildTenorLedger( _
                result, _
                outputRow, _
                curveData, _
                dateCol, _
                tenorColumns(tenorIndex), _
                CStr(tenors(tenorIndex)), _
                requestedStart, _
                analysisEndDate, _
                startingPrincipal, _
                initialDates(tenorIndex), _
                initialRates(tenorIndex), _
                buildError) Then

            r_DailyAccrual = ErrorResult(buildError)
            Exit Function
        End If
    Next tenorIndex

    If outputRow <> totalRows + 1 Then
        r_DailyAccrual = ErrorResult("Internal row-count reconciliation failed.")
        Exit Function
    End If

    r_DailyAccrual = result
    Exit Function

CleanFail:
    r_DailyAccrual = ErrorResult("VBA error " & Err.Number & ": " & Err.Description)
End Function

'====================================================================
' Build one complete daily ledger for one tenor.
'====================================================================

Private Function BuildTenorLedger( _
    ByRef result As Variant, _
    ByRef outputRow As Long, _
    ByRef curveData As Variant, _
    ByVal dateCol As Long, _
    ByVal rateCol As Long, _
    ByVal tenor As String, _
    ByVal requestedStart As Date, _
    ByVal analysisEndDate As Date, _
    ByVal initialPrincipal As Double, _
    ByVal initialStart As Date, _
    ByVal initialRate As Double, _
    ByRef errorMessage As String) As Boolean

    Dim tenorMonths As Long
    Dim transactionNumber As Long
    Dim currentStart As Date
    Dim targetRoll As Date
    Dim actualRoll As Date
    Dim nextRateDate As Date
    Dim rateDate As Date
    Dim accrualDate As Date
    Dim accrualLastDate As Date

    Dim currentRate As Double
    Dim nextRate As Double
    Dim principal As Double
    Dim dailyInterest As Double
    Dim accruedInterest As Double
    Dim totalTransactionInterest As Double

    Dim transactionDays As Long
    Dim dayNumber As Long
    Dim isCompleted As Boolean
    Dim found As Boolean

    errorMessage = vbNullString
    tenorMonths = TenorToMonths(tenor)

    principal = initialPrincipal
    currentStart = initialStart
    currentRate = initialRate
    rateDate = initialStart
    transactionNumber = 1

    Do While currentStart <= analysisEndDate

        targetRoll = 0
        actualRoll = 0
        nextRateDate = 0
        nextRate = 0
        isCompleted = False
        accruedInterest = 0

        If tenor = "ON" Then
            'ON rolls on the next date carrying a valid ON observation.
            found = GetNextPointAfter( _
                curveData, dateCol, rateCol, currentStart, nextRateDate, nextRate)

            If found Then
                targetRoll = nextRateDate
                If nextRateDate <= analysisEndDate Then
                    actualRoll = nextRateDate
                    isCompleted = True
                End If
            End If

        Else
            'Monthly targets always reference the original requested start.
            targetRoll = AddMonthsAnchored( _
                requestedStart, transactionNumber * tenorMonths)

            If targetRoll <= analysisEndDate Then
                found = GetPointOnOrBefore( _
                    curveData, dateCol, rateCol, targetRoll, actualRoll, nextRate)

                If Not found Then
                    errorMessage = "No valid " & tenor & _
                        " rate exists on or before target roll date " & _
                        Format$(targetRoll, "yyyy-mm-dd") & "."
                    Exit Function
                End If

                If actualRoll <= currentStart Then
                    errorMessage = "Adjusted " & tenor & " roll date " & _
                        Format$(actualRoll, "yyyy-mm-dd") & _
                        " is not after transaction start " & _
                        Format$(currentStart, "yyyy-mm-dd") & "."
                    Exit Function
                End If

                nextRateDate = actualRoll
                isCompleted = True
            End If
        End If

        If isCompleted Then
            accrualLastDate = DateAdd("d", -1, actualRoll)
            transactionDays = DateDiff("d", currentStart, actualRoll)
        Else
            accrualLastDate = analysisEndDate
            transactionDays = DateDiff("d", currentStart, analysisEndDate) + 1
        End If

        If transactionDays <= 0 Then
            errorMessage = "A " & tenor & " transaction has zero or negative accrual days."
            Exit Function
        End If

        'Rates such as 1.54 represent 1.54%, therefore divide by 100.
        totalTransactionInterest = _
            principal * (currentRate / 100#) * transactionDays / 360#

        dayNumber = 0

        accrualDate = currentStart
        Do While accrualDate <= accrualLastDate
            dayNumber = dayNumber + 1
            dailyInterest = principal * (currentRate / 100#) / 360#
            accruedInterest = accruedInterest + dailyInterest

            result(outputRow, 1) = accrualDate
            result(outputRow, 2) = tenor
            result(outputRow, 3) = transactionNumber
            result(outputRow, 4) = requestedStart
            result(outputRow, 5) = currentStart

            If isCompleted Then
                result(outputRow, 6) = actualRoll
            Else
                result(outputRow, 6) = Empty
            End If

            If targetRoll <> 0 Then
                result(outputRow, 7) = targetRoll
            Else
                result(outputRow, 7) = Empty
            End If

            result(outputRow, 8) = rateDate
            result(outputRow, 9) = currentRate
            result(outputRow, 10) = principal
            result(outputRow, 11) = dailyInterest
            result(outputRow, 12) = accruedInterest

            If isCompleted Then
                result(outputRow, 13) = totalTransactionInterest
            Else
                result(outputRow, 13) = Empty
            End If

            result(outputRow, 14) = principal + accruedInterest
            result(outputRow, 15) = dayNumber
            result(outputRow, 16) = transactionDays

            If accrualDate = currentStart Then
                If transactionNumber = 1 Then
                    result(outputRow, 17) = "START"
                Else
                    result(outputRow, 17) = "ROLL"
                End If
            Else
                result(outputRow, 17) = vbNullString
            End If

            If isCompleted Then
                result(outputRow, 18) = "COMPLETED"
            Else
                result(outputRow, 18) = "OPEN AT ANALYSIS END"
            End If

            outputRow = outputRow + 1
            accrualDate = DateAdd("d", 1, accrualDate)
        Loop

        If Not isCompleted Then Exit Do

        'Interest is always reinvested.
        principal = principal + totalTransactionInterest
        currentStart = actualRoll
        currentRate = nextRate
        rateDate = nextRateDate
        transactionNumber = transactionNumber + 1
    Loop

    BuildTenorLedger = True
End Function

'====================================================================
' Output headers
'====================================================================

Private Sub WriteHeaders(ByRef result As Variant)
    result(1, 1) = "Accrual Date"
    result(1, 2) = "Tenor"
    result(1, 3) = "Transaction ID"
    result(1, 4) = "Requested Analysis Start"
    result(1, 5) = "Transaction Start Date"
    result(1, 6) = "Transaction End / Actual Roll Date"
    result(1, 7) = "Target Roll Date"
    result(1, 8) = "Rate Date"
    result(1, 9) = "Rate Used (%)"
    result(1, 10) = "Opening Notional"
    result(1, 11) = "Daily Interest"
    result(1, 12) = "Accrued Interest in Period"
    result(1, 13) = "Total Transaction Interest"
    result(1, 14) = "Accrued Balance"
    result(1, 15) = "Day Number in Transaction"
    result(1, 16) = "Calendar Days in Transaction"
    result(1, 17) = "Roll Flag"
    result(1, 18) = "Status"
End Sub

'====================================================================
' Curve validation and lookup helpers
'====================================================================

Private Function FindHeaderColumn( _
    ByRef data As Variant, _
    ByVal requiredHeader As String) As Long

    Dim col As Long
    Dim headerText As String

    requiredHeader = NormalizeHeader(requiredHeader)

    For col = 1 To UBound(data, 2)
        headerText = NormalizeHeader(CStr(data(1, col)))
        If headerText = requiredHeader Then
            FindHeaderColumn = col
            Exit Function
        End If
    Next col

    FindHeaderColumn = 0
End Function

Private Function ValidateCurveData( _
    ByRef data As Variant, _
    ByVal dateCol As Long, _
    ByRef tenorColumns() As Long) As String

    Dim row As Long
    Dim tenorIndex As Long
    Dim curveDate As Date
    Dim dateKey As String
    Dim datesSeen As Object
    Dim validRateCount(0 To 4) As Long
    Dim cellValue As Variant

    Set datesSeen = CreateObject("Scripting.Dictionary")

    For row = 2 To UBound(data, 1)
        cellValue = data(row, dateCol)

        If HasValue(cellValue) Then
            If Not TryGetExcelDate(cellValue, curveDate) Then
                ValidateCurveData = "Invalid curve date in data row " & row & "."
                Exit Function
            End If

            dateKey = CStr(CLng(curveDate))
            If datesSeen.Exists(dateKey) Then
                ValidateCurveData = "Duplicate curve date found: " & _
                    Format$(curveDate, "yyyy-mm-dd") & "."
                Exit Function
            End If
            datesSeen.Add dateKey, True

            For tenorIndex = 0 To 4
                cellValue = data(row, tenorColumns(tenorIndex))

                If HasValue(cellValue) Then
                    If Not IsNumeric(cellValue) Then
                        ValidateCurveData = "Non-numeric rate in data row " & row & "."
                        Exit Function
                    End If
                    validRateCount(tenorIndex) = validRateCount(tenorIndex) + 1
                End If
            Next tenorIndex
        End If
    Next row

    If datesSeen.Count = 0 Then
        ValidateCurveData = "The curve range contains no valid dates."
        Exit Function
    End If

    For tenorIndex = 0 To 4
        If validRateCount(tenorIndex) = 0 Then
            ValidateCurveData = "The curve contains no valid rates for column " & _
                TenorNameByIndex(tenorIndex) & "."
            Exit Function
        End If
    Next tenorIndex

    ValidateCurveData = vbNullString
End Function

Private Function GetPointOnOrBefore( _
    ByRef data As Variant, _
    ByVal dateCol As Long, _
    ByVal rateCol As Long, _
    ByVal targetDate As Date, _
    ByRef foundDate As Date, _
    ByRef foundRate As Double) As Boolean

    Dim row As Long
    Dim curveDate As Date
    Dim bestDate As Date
    Dim hasPoint As Boolean

    hasPoint = False
    bestDate = 0

    For row = 2 To UBound(data, 1)
        If TryGetExcelDate(data(row, dateCol), curveDate) Then
            If IsNumeric(data(row, rateCol)) And HasValue(data(row, rateCol)) Then
                If curveDate <= targetDate Then
                    If Not hasPoint Or curveDate > bestDate Then
                        bestDate = curveDate
                        foundRate = CDbl(data(row, rateCol))
                        hasPoint = True
                    End If
                End If
            End If
        End If
    Next row

    If hasPoint Then foundDate = bestDate
    GetPointOnOrBefore = hasPoint
End Function

Private Function GetNextPointAfter( _
    ByRef data As Variant, _
    ByVal dateCol As Long, _
    ByVal rateCol As Long, _
    ByVal currentDate As Date, _
    ByRef foundDate As Date, _
    ByRef foundRate As Double) As Boolean

    Dim row As Long
    Dim curveDate As Date
    Dim bestDate As Date
    Dim hasPoint As Boolean

    hasPoint = False
    bestDate = 0

    For row = 2 To UBound(data, 1)
        If TryGetExcelDate(data(row, dateCol), curveDate) Then
            If IsNumeric(data(row, rateCol)) And HasValue(data(row, rateCol)) Then
                If curveDate > currentDate Then
                    If Not hasPoint Or curveDate < bestDate Then
                        bestDate = curveDate
                        foundRate = CDbl(data(row, rateCol))
                        hasPoint = True
                    End If
                End If
            End If
        End If
    Next row

    If hasPoint Then foundDate = bestDate
    GetNextPointAfter = hasPoint
End Function

'====================================================================
' Date helpers
'====================================================================

Private Function TryGetExcelDate( _
    ByVal value As Variant, _
    ByRef resultDate As Date) As Boolean

    Dim serialNumber As Double
    Dim textValue As String

    On Error GoTo InvalidDate

    If IsError(value) Or IsEmpty(value) Then Exit Function

    'Range.Value2 returns real Excel dates as numeric serials.
    If IsNumeric(value) Then
        serialNumber = CDbl(value)

        'Valid modern Excel serial range. The integer date is used;
        'any time component is removed.
        If serialNumber >= 1 And serialNumber < 2958466# Then
            resultDate = DateValue(DateSerial(1899, 12, 30) + Fix(serialNumber))
            TryGetExcelDate = True
            Exit Function
        End If
    End If

    textValue = Trim$(CStr(value))
    If Len(textValue) = 0 Then Exit Function

    If IsDate(textValue) Then
        resultDate = DateValue(CDate(textValue))
        TryGetExcelDate = True
        Exit Function
    End If

InvalidDate:
    TryGetExcelDate = False
End Function

Private Function AddMonthsAnchored( _
    ByVal anchorDate As Date, _
    ByVal monthsToAdd As Long) As Date

    Dim targetFirstDay As Date
    Dim anchorLastDay As Long
    Dim targetLastDay As Long
    Dim targetDay As Long

    targetFirstDay = DateSerial(Year(anchorDate), Month(anchorDate) + monthsToAdd, 1)
    anchorLastDay = Day(DateSerial(Year(anchorDate), Month(anchorDate) + 1, 0))
    targetLastDay = Day(DateSerial(Year(targetFirstDay), Month(targetFirstDay) + 1, 0))

    If Day(anchorDate) = anchorLastDay Then
        targetDay = targetLastDay
    ElseIf Day(anchorDate) > targetLastDay Then
        targetDay = targetLastDay
    Else
        targetDay = Day(anchorDate)
    End If

    AddMonthsAnchored = DateSerial( _
        Year(targetFirstDay), Month(targetFirstDay), targetDay)
End Function

'====================================================================
' General helpers
'====================================================================

Private Function NormalizeHeader(ByVal value As String) As String
    Dim textValue As String

    textValue = UCase$(Trim$(value))
    textValue = Replace(textValue, " ", vbNullString)
    textValue = Replace(textValue, "-", vbNullString)
    textValue = Replace(textValue, "_", vbNullString)

    Select Case textValue
        Case "O/N", "OVERNIGHT"
            NormalizeHeader = "ON"
        Case "1MONTH", "01M"
            NormalizeHeader = "1M"
        Case "2MONTH", "02M"
            NormalizeHeader = "2M"
        Case "3MONTH", "03M"
            NormalizeHeader = "3M"
        Case "6MONTH", "06M"
            NormalizeHeader = "6M"
        Case Else
            NormalizeHeader = textValue
    End Select
End Function

Private Function TenorToMonths(ByVal tenor As String) As Long
    Select Case UCase$(Trim$(tenor))
        Case "ON": TenorToMonths = 0
        Case "1M": TenorToMonths = 1
        Case "2M": TenorToMonths = 2
        Case "3M": TenorToMonths = 3
        Case "6M": TenorToMonths = 6
        Case Else: TenorToMonths = -1
    End Select
End Function

Private Function TenorNameByIndex(ByVal tenorIndex As Long) As String
    Select Case tenorIndex
        Case 0: TenorNameByIndex = "ON"
        Case 1: TenorNameByIndex = "1M"
        Case 2: TenorNameByIndex = "2M"
        Case 3: TenorNameByIndex = "3M"
        Case 4: TenorNameByIndex = "6M"
        Case Else: TenorNameByIndex = "UNKNOWN"
    End Select
End Function

Private Function HasValue(ByVal value As Variant) As Boolean
    If IsError(value) Or IsEmpty(value) Then
        HasValue = False
    Else
        HasValue = (Len(Trim$(CStr(value))) > 0)
    End If
End Function

Private Function ErrorResult(ByVal message As String) As Variant
    Dim errorArray(1 To 2, 1 To 1) As Variant

    errorArray(1, 1) = "ERROR"
    errorArray(2, 1) = message
    ErrorResult = errorArray
End Function
